// A non-zero-width bitfield always increases the size by the full
// size of the underlying type. Even if we are in a packed context.
//
// ```c,msvc,msvc-tc-0040
// #pragma pack(1)
//
// struct S {
//         char c;
//         int :1;
// };
//
// static void f(void) {
//         static_assert(sizeof(struct S) == 5, "");
// }
// ```
S = @pragma_pack(1) struct {
    c char,
    _ int:1,
}
