// If all fields in a union have size 0, the size of the whole enum is set to ...
// ... 4 bytes if it contains only bitfields.
//
// ```c,msvc,msvc-tc-0025
// union X {
//         long long :0;
// };
//
// union Y {
//         char :0;
// };
//
// static void f(void) {
//         static_assert(sizeof(union X) == 4, "");
//         static_assert(sizeof(union Y) == 4, "");
// }
// ```
X = union {
    _ long long:0,
}
Y = union {
    _ char:0,
}
