// If all fields in a union have size 0, the size of the whole enum is set to ...
// ... its alignment if it contains at least one non-bitfield.
//
// ```c,msvc,msvc-tc-0024
// union X {
//         long long b[0];
// };
//
// union Y {
//         long long :0;
//         char b[0];
// };
//
// static void f(void) {
//         static_assert(sizeof(union X) == 8, "");
//         static_assert(sizeof(union Y) == 1, "");
// }
// ```
X = union {
    b []long long,
}

Y = union {
    _ long long:0,
    b []char,
}
