AlignedStruct = @align(16) struct {
    x int,
}
AlignedInt = @align(16) typedef int

A = @pragma_pack(2) struct {
    a [1]AlignedStruct,
}

B = @pragma_pack(2) struct {
    b char,
    a [1]AlignedInt,
}

C = @pragma_pack(2) struct {
    c char,
    a []AlignedInt,
}
